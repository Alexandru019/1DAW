		------Comandos de Usuarios------
useradd
groupadd nomgrupo      crea un grupo
useradd -g nombregrupo nombreusuario    añade un usuario a un grupo
cat /etc/passwd          para ver usuarios
useradd nombrequequieras -m	obliga a crear la carpeta personal
useradd nombrequequieras -m -d /home/nombrequequieras	obliga a crear la carpeta y le pone el nombre que quieras
userdel nombreusuario             elimina usuarios
userdel -r nombreusuario        elimina usuario y carpeta personal
chmod 123 nomfichero           cambia los permisos
chown nomusu nomfichero        cambia el usuario propietario
chgrp nomgrp nomfichero        cambia el grupo proietario
